\begin{frame}{Coin-flipping Experiment \alt<2>{B}{A}}
	\begin{enumerate}
		\item	Participant comes into lab
		\item	Fills out forms
			\begin{itemize}
			\item	consent
			\item	demographic info
			\end{itemize}
		\item	Start video recording(s)
		\item	\alt<2>{Experimenter}{Participant} flips coin 10 times, \alt<2>{participant}{experimenter} calls
		\item	\alt<1>{Experimenter}{Participant} flips coin 10 times, \alt<1>{participant}{experimenter} calls
		\item	Stop video recording
		\item	Databrary release form
	\end{enumerate}
\end{frame}
